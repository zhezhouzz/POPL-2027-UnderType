\section{Operational Semantics}\label{sec:tech:semantics}

\begin{figure}[h!]
 {\small
  {\normalsize
\begin{flalign*}
 &\text{\textbf{Operational Semantics }} & \fbox{$e \hookrightarrow e$}
\end{flalign*}
}
\\ \
\begin{prooftree}
\hypo{op\ \overline{v} \Downarrow v_y }
\infer1[\textsc{StAppOp}]{
\zlet{y}{op\ \overline{v}}{e} \hookrightarrow e[y\mapsto v_y]
}
\end{prooftree}
\\ \ \\ \ \\
\begin{prooftree}
\hypo{e_1 \hookrightarrow e_1'}
\infer1[\textsc{StLetE1}]{
\zlet{y}{e_1}{e_2} \hookrightarrow \zlet{y}{e_1'}{e_2}
}
\end{prooftree}
\quad
\begin{prooftree}
\hypo{}
\infer1[\textsc{StLetE2}]{
\zlet{y}{v}{e} \hookrightarrow e[y\mapsto v]
}
\end{prooftree}
\\ \ \\ \ \\
\begin{prooftree}
\hypo{}
\infer1[\textsc{StLetAppLam}]{
\zlet{y}{\zzlam{x}{e_1}\ v_x}{e_2} \hookrightarrow \zlet{y}{e_1[x\mapsto v_x]}{e_2}
}
\end{prooftree}
\\ \ \\ \ \\
\begin{prooftree}
\hypo{}
\infer1[\textsc{StLetAppFix}]{
\zlet{y}{\zzfix{f}{x}{e_1}\ v_x}{e_2} \hookrightarrow
\zlet{y}{(\zzlam{f}{e_1[x\mapsto v_x]}) \ (\zzfix{f}{x}{e_1})}{e_2}
}
\end{prooftree}
\\ \ \\ \ \\
\begin{prooftree}
\hypo{}
\infer1[\textsc{StMatch}]{
\match{d_i \ \overline{v_j}} \overline{d_i\ \overline{y_j} \to e_i} \hookrightarrow e_i[\overline{y_j \mapsto v_j}]
}
\end{prooftree}
}
\caption{Small Step Operational Semantics}
    \label{fig:semantics}
\end{figure}

The small-step operational semantics of \langname{} are shown in
\autoref{fig:semantics}. The rules are parameterized over the
semantics of built-in operators
$\primop\; \overline{v} \Downarrow v_y$.
